    private function managePicture($files){
        // Préparation du futur nom de l'image pour qu'il soit unique.
        // On récupère l'extension du fichier téléchargé à l'aide de la méthode native pathinfo
        $ext = pathinfo($files["picture"]["name"], PATHINFO_EXTENSION);
        // On crée un nom pour l'image en utilisant la méthode native uniqueid()
        // https://www.php.net/manual/fr/function.uniqid.php
        $name = "article-".uniqid().'.'.$ext;
        // On déplace l'image du dossier temporaire du serveur pour la mettre dans le dossier image du site (front)
        move_uploaded_file($files['picture']['tmp_name'],'../../images/'.$name);
        // On pourrait aller plus loin en prenant en charge le redimessionement de l'image
        // https://www.php.net/manual/fr/book.image.php sinon pour Symfony cf liip Image sur packagist.org
        // On renvoit le nom de l'image
        return $name;
    }
